# WhatsApp Business Webhook Configuration
# Copy this file to .env.local and fill in your actual values

# Webhook Verification Token
# Set this to any secure random string of your choice
# You'll need to enter the same value in the Meta App Dashboard
WEBHOOK_VERIFY_TOKEN=your_secure_verification_token_here

# WhatsApp App Secret
# Get this from Meta App Dashboard → Settings → Basic → App Secret
# This is used to validate webhook signatures for security
WHATSAPP_APP_SECRET=your_app_secret_from_meta_dashboard

# Supabase Configuration
# Get these from your Supabase project dashboard → Settings → API
NEXT_PUBLIC_SUPABASE_URL=your_supabase_project_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key
